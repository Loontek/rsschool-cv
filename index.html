Code Example

A function that finds Fibonacci numbers. Takes the step number as an argument 


'use strict'

function fib(n) {
	if (n === 1 || n === 2) {
		return 1;
	}
	return fib(n - 1) + fib(n - 2);
}

console.log(fib(7)) //  13
